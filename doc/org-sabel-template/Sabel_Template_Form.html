Sabel_Template_Form


Example :  
  1.    <? $form = new Sabel_Template_Form($model->schema(), (isset($errors)) ? $errors : null) ?>
    2. 

  2.    <? $form->hidden(array('shop_id', 'users_id')) ?>
    3. 

  3.    <?= $form->startTag(uri(array('action' => 'save', 'id' => $model->id)), 'POST') ?>
    4. 

  4.    <? foreach ($form as $f) : ?>
    5. 

  5.      <?= $f->write("{$f->name()}<br />", "<br /><br />") ?>
    6. 

  6.    <? endforeach ?>
    7. 

  7.    <?= $form->submitTag('save') ?>
    8. 

  8.    <?= $form->endTag() ?>
    9.
